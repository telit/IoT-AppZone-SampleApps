253 /* block lock register defines
254 * Register Bit Address
255 * | 7 | 6 | 5 | 4 | 3 | 2 | 1 | 0 |
256 * |----------------------------------------------------------|
257 * | BRWD | Reversed | BP2 | BP1 | BP0 | INV | CMP | Reversed |
258 *
259 * Block Lock Register Block Protection Bits
260 */
261 #define LOCK_CMP (0x02)
262 #define LOCK_INV (0x04)
263 #define LOCK_BP0 (0x08)
264 #define LOCK_BP1 (0x10)
265 #define LOCK_BP2 (0x20)
266 
267  /* CMP INV BP2 BP1 BP0 Protected Portion*/
268 #define BL_ALL_UNLOCKED (0x00) /* x x 0 0 0 All unlocked */
269 #define BL_U_1_64_LOCKED (LOCK_BP0) /* 0 0 0 0 1 Upper 1/64 locked */
270 #define BL_U_1_32_LOCKED (LOCK_BP1) /* 0 0 0 1 0 Upper 1/32 locked */
271 #define BL_U_1_16_LOCKED (LOCK_BP1 | LOCK_BP0) /* 0 0 0 1 1 Upper 1/16 locked */
272 #define BL_U_1_8_LOCKED (LOCK_BP2) /* 0 0 1 0 0 Upper 1/8 locked */
273 #define BL_U_1_4_LOCKED (LOCK_BP2 | LOCK_BP0) /* 0 0 1 0 1 Upper 1/4 locked */
274 #define BL_U_1_2_LOCKED (LOCK_BP2 | LOCK_BP1) /* 0 0 1 1 0 Upper 1/2 locked */
275 #define BL_ALL_LOCKED (LOCK_CMP | LOCK_INV | LOCK_BP2 | LOCK_BP1 | LOCK_BP0) /* x x 1 1 1 All Locked (default) */
276 #define BL_L_1_64_LOCKED (LOCK_INV | LOCK_BP0) /* 0 1 0 0 1 Lower 1/64 locked */
277 #define BL_L_1_32_LOCKED (LOCK_INV | LOCK_BP1) /* 0 1 0 1 0 Lower 1/32 locked */
278 #define BL_L_1_16_LOCKED (LOCK_INV | LOCK_BP1 | LOCK_BP0) /* 0 1 0 1 1 Lower 1/16 locked */
279 #define BL_L_1_8_LOCKED (LOCK_INV | LOCK_BP2) /* 0 1 1 0 0 Lower 1/8 locked */
280 #define BL_L_1_4_LOCKED (LOCK_INV | LOCK_BP2 | LOCK_BP0) /* 0 1 1 0 1 Lower 1/4 locked */
281 #define BL_L_1_2_LOCKED (LOCK_INV | LOCK_BP2 | LOCK_BP1) /* 0 1 1 1 0 Lower 1/2 locked */
282 #define BL_L_63_64_LOCKED (LOCK_CMP | LOCK_BP0) /* 1 0 0 0 1 Lower 63/64 locked */
283 #define BL_L_31_32_LOCKED (LOCK_CMP | LOCK_BP1) /* 1 0 0 1 0 Lower 31/32 locked */
284 #define BL_L_15_16_LOCKED (LOCK_CMP | LOCK_BP1 | LOCK_BP0) /* 1 0 0 1 1 Lower 15/16 locked */
285 #define BL_L_7_8_LOCKED (LOCK_CMP | LOCK_BP2) /* 1 0 1 0 0 Lower 7/8 locked */
286 #define BL_L_3_4_LOCKED (LOCK_CMP | LOCK_BP2 | LOCK_BP0) /* 1 0 1 0 1 Lower 3/4 locked */
287 #define BL_BLOCK0_LOCKED (LOCK_CMP | LOCK_BP2 | LOCK_BP1) /* 1 0 1 1 0 Block 0 locked */
