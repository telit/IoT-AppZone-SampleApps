69 static INT32 ftp_debug_hk(AZX_FTP_DEBUG_HOOK_LEVELS_E level, const CHAR *function,
70  const CHAR *file, INT32 line, const CHAR *fmt, ...)
71 {
72  char buf[512];
73  int bufSize = sizeof(buf);
74  va_list arg;
75  INT32 offset = 0;
76  UINT32 now = get_uptime();
77 
78  memset(buf,0,bufSize);
79 
80  switch(level)
81  {
82  case AZX_FTP_DEBUG_HOOK_ERROR:
83  offset = sprintf(buf, "%5u.%03u %6s - %15s:%-4d - %32s - ",
84  now / 1000, now % 1000,
85  "ERR - ",
86  get_file_title(file), line,
87  function);
88  break;
89  break;
90  case AZX_FTP_DEBUG_HOOK_INFO:
91  offset = 0;
92  break;
93  case AZX_FTP_DEBUG_HOOK_DEBUG:
94  offset = sprintf(buf, "%5u.%03u %6s - %15s:%-4d - %32s - ",
95  now / 1000, now % 1000,
96  "DBG - ",
97  get_file_title(file), line,
98  function);
99  break;
100  default:
101  offset = 0;
102  break;
103  }
104  va_start(arg, fmt);
105  vsnprintf(buf + offset, bufSize-offset, fmt, arg);
106  va_end(arg);
107 
108  return AZX_LOG_INFO(buf);
109 }
