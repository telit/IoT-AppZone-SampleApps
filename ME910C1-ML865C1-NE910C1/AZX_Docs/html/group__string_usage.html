Trim Macros

These macros can be used to trim strings removing quotes

#define AZX_TRIM_LEADING_QUOTES(a)   if('\"'==a[0]){a++;}
 
#define AZX_TRIM_TRAILING_QUOTES(a)   if('\"'==a[strlen(a)-1]){a[strlen(a)-1]=0;}
 
#define AZX_TRIM_LEADING_AND_TRAILING_QUOTES(a)   TRIM_LEADING_QUOTES(a)TRIM_TRAILING_QUOTES(a)
